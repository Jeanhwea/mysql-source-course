* join3_01
- [[file:join3_01_opt.sql][sql]] | [[file:join3_01_opt.sql.json][opt_trace]]
#+BEGIN_SRC sql
  select
    e.first_name, e.last_name, a.from_date
  from
    dept_manager a
    join departments d on a.dept_no = d.dept_no
    join employees e on a.emp_no = e.emp_no
  where
    d.dept_no = 'd001';
#+END_SRC
